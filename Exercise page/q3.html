Write a simple DHTML program that changes the properties of the table

Copied!

Solution:

HTML:

<!DOCTYPE html> <head> <title>Change Table</title> <link rel="stylesheet" href="q1.css"> </head> <body> <table id = "table" align = "center"> <tr id="row1"> <td>A</td> <td>B</td> <td>C</td> </tr> <tr id="row2"> <td>D</td> <td>E</td> <td>F</td> </tr> <tr id="row3"> <td>G</td> <td>H</td> <td>I</td> </tr> <tr id="row4"> <td id = "myTd">J</td> <td>K</td> <td>L</td> </tr> </table><br> <h2>Make changes to the table:</h2> <button onclick = "chwidth()">Change width of table border</button><br><br> <button onclick = "caption()">Create caption for the table</button><br><br> <button onclick = "addrow()">Add new row to table</button><br><br> <button onclick = "addcell()">Add new cell to table</button><br><br> <button onclick = "chcell()">Change the content of a table cell in third row</button><br><br> <button onclick = "chcolspan()">Change the colspan of a table row</button><br><br> <script> function chwidth(){ document.getElementById("table").style.borderWidth = "5px"; } function caption(){ var table = document.getElementById("table").createCaption(); table.innerHTML = "<b>My Table</b>"; } function addrow(){ var table = document.getElementById("table"); var row = table.insertRow(4); var cell1 = row.insertCell(0); var cell2 = row.insertCell(1); var cell3 = row.insertCell(2); cell1.innerHTML = "NEW CELL1"; cell2.innerHTML = "NEW CELL2"; cell3.innerHTML = "NEW CELL3"; } function addcell(){ var row = document.getElementById("row1"); var x = row.insertCell(3); x.innerHTML = "New cell"; } function chcell(){ var table = document.getElementById("table"); table.rows[2].cells[0].innerHTML = "Text modified"; } function chcolspan(){ document.getElementById("myTd").colSpan = "2"; } </script> </body>

CSS:

body{ margin :20px; text-align: center; margin-top: 50px; } table , td{ border-style: solid; border-color: crimson; border-collapse: collapse; border-width: 2px; } td{ padding: 10px; text-align: left; } #row1{ background-color : #51A7E2; } #row2{ background-color : #ffa500; } #row3{ background-color : #32cd32; } #row4{ background-color : #dea5a4; }
